corvee_enabled = {
	set_variable = {
		name = corvee_pops_var
		value = corvee_pops
	}
	every_scope_state = {
		limit = {
			is_incorporated = yes
			has_building = building_construction_sector
		}
		every_scope_pop = {
			limit = {
				OR = {
					is_pop_type = peasants
					is_pop_type = farmers
					is_pop_type = clerks
					is_pop_type = shopkeepers
				}
				OR = {
					is_employed = no
					pop_employment_building_group = bg_subsistence_agriculture
				}
				this.culture = {
					is_primary_culture_of = ROOT
				}
			}
			change_poptype = pop_type:laborers
			#change_variable = {
			#	name = corvee_pops
			#	add = 10
			#}
			set_pop_qualifications = {
				pop_type = laborers
				qualifications = 0.75
			}
		}
	}
	
}
