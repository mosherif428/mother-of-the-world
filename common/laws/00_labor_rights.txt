# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_corvee = {						# clerks, farmers, peasants, shopkeepers --> laborers
    group = lawgroup_labor_rights
    icon = "gfx/interface/icons/law_icons/corvee.dds"
    progressiveness = -10

    modifier = {
		country_construction_add = 50 # corvee_pops
		state_radicals_from_sol_change_mult = 1.0
		state_poor_standard_of_living_add = -2
		state_middle_standard_of_living_add = -1
	}

	possible_political_movements = {
		law_no_workers_rights
		law_regulatory_bodies
	}

	pop_support = {
		value = 0

		add = {
			desc = "POP_FORCED_LABOR"		
			if = {
				limit = { 
					NOT = { strata = rich }
					OR = {
						is_pop_type = peasants
						is_pop_type = farmers
						is_pop_type = clerks
						is_pop_type = shopkeepers
						is_pop_type = laborers
					}
				}
				add = {
					value = literacy_rate
					multiply = -2
				}
			}			
		}
	}

	ai_enact_weight_modifier = {
		value = 0
		if = {
			limit = {
				any_scope_pop = {
					count < 100
				}
			}
			add = -99999
		}
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
		if = {
			limit = {
				has_journal_entry = je_muhammad_alis_corvee
			}
			add = 1000
		}
	}

	on_activate = {
		corvee_enabled = yes
		set_variable = {
			name = corvee_enacted
		}
	}

	pop_support = {
		value = 0
		add = {
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 0.2
			}
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 0.2
			}
			if = {
				limit = { 
					is_pop_type = academics
				}
				value = -0.2
			}
			if = {
				limit = { 
					is_pop_type = laborers
				}
				value = -1.0
			}
			if = {
				limit = {
					is_pop_type = peasants
				}
				value = -0.5
			}
			if = {
				limit = {
					is_pop_type = farmers
				}
				value = -0.5
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				value = -0.5
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				value = -0.5
			}
		}
	}

	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_slaver
		}
	}
}
